TASK_NM="${taskNo}"; STOP_CMD="${service.stop}"; if [ -z "$TASK_NM" ]; then echo "ERROR : TASK NAME is required"; exit 1; elif [ -d "$TASK_NM" ]; then echo "ERROR : DIRECTORY $TASK_NM already exists"; exit 1; fi; mkdir "$TASK_NM"; IFS=","; if [ -n "$STOP_CMD" ]; then for sc in $STOP_CMD; do eval "$sc" 1>&2 2>/dev/null; rc=$?; [ $rc -ne 0 ] && echo "WARN : STOP COMMAND not able to execute, Code : $rc"; done; fi; exit 0;
